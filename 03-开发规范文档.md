## 一、命名规范

### 1.1 文件命名

| 类型 | 规范 | 示例 |
|------|------|------|
| **页面文件** | 小写 + 中划线 | `game.js`, `game.wxml`, `game.wxss` |
| **组件文件** | 小写 + 中划线 | `card-item.js`, `slot-bar.js` |
| **工具函数** | 小写 + 中划线 | `card-generator.js`, `coverage-calc.js` |
| **配置文件** | 小写 + 中划线 | `level-config.js`, `card-config.js` |
| **资源文件** | 小写 + 下划线 | `sunflower.png`, `water_drop.png` |
| **类文件** | 大驼峰 | `GameState.js`, `Card.js`, `AdManager.js` |

### 1.2 变量命名

```javascript
Component({
  /**
   * 组件属性
   */
  properties: {
    cardData: {
      type: Object,
      value: {}
    },
    clickable: {
      type: Boolean,
      value: true
    }
  },

  /**
   * 组件数据
   */
  data: {
    animationData: null
  },

  /**
   * 组件方法
   */
  methods: {
    onTap() {
      if (this.data.clickable) {
        this.triggerEvent('tap', { cardId: this.data.cardData.id });
      } else {
        this.shake();
      }
    },

    shake() {
      const animation = wx.createAnimation({
        duration: 300,
        timingFunction: 'ease-in-out'
      });
      
      animation.translateX(-5).step()
        .translateX(5).step()
        .translateX(0).step();
      
      this.setData({
        animationData: animation.export()
      });
    }
  }
});
```

```javascript
// utils/api.js
const BASE_URL = 'https://api.example.com';

/**
 * 统一请求方法
 */
function request(options) {
  return new Promise((resolve, reject) => {
    wx.request({
      url: BASE_URL + options.url,
      method: options.method || 'GET',
      data: options.data,
      header: {
        'Content-Type': 'application/json',
        ...options.header
      },
      success: (res) => {
        if (res.statusCode === 200) {
          resolve(res.data);
        } else {
          reject(new Error(`请求失败：${res.statusCode}`));
        }
      },
      fail: (err) => {
        reject(err);
      }
    });
  });
}

/**
 * GET 请求
 */
function get(url, data) {
  return request({ url, method: 'GET', data });
}

/**
 * POST 请求
 */
function post(url, data) {
  return request({ url, method: 'POST', data });
}

module.exports = { request, get, post };
```

## 七、错误处理规范

### 7.1 统一错误处理

```javascript
// utils/error-handler.js

/**
 * 错误类型
 */
const ErrorType = {
  NETWORK: 'NETWORK_ERROR',
  DATA: 'DATA_ERROR',
  GAME: 'GAME_ERROR',
  UNKNOWN: 'UNKNOWN_ERROR'
};

/**
 * 统一错误处理
 */
function handleError(error, context = '') {
  console.error(`[${context}] 错误:`, error);
  
  // 上报错误
  wx.reportAnalytics('error', {
    type: error.type || ErrorType.UNKNOWN,
    message: error.message,
    context,
    timestamp: Date.now()
  });
  
  // 显示错误提示
  wx.showToast({
    title: getErrorMessage(error),
    icon: 'none',
    duration: 2000
  });
}

/**
 * 获取错误消息
 */
function getErrorMessage(error) {
  switch (error.type) {
    case ErrorType.NETWORK:
      return '网络错误，请检查连接';
    case ErrorType.DATA:
      return '数据错误，请重试';
    case ErrorType.GAME:
      return '游戏错误，请重新开始';
    default:
      return '发生错误，请稍后重试';
  }
}

module.exports = { handleError, ErrorType };
```
