interface Card {
  id: string;                  // 唯一标识 "card_001"
  type: number;                // 牌型 ID (1-12)
  typeId: string;              // 牌型字符串 "sheep_head"

  // 屏幕坐标 (像素)
  x: number;                   // X 坐标
  y: number;                   // Y 坐标
  zIndex: number;              // CSS z-index

  isCovered: boolean;          // 是否被覆盖
  isSelected: boolean;         // 是否被选中
  
  // 视觉
  rotation: number;            // 旋转角度 (-5 到 5 度随机)
  scale: number;               // 缩放比例 (0.95-1.05 随机)
}

  HIDDEN = 0,                  // 不可见
  COVERED = 1,                 // 被覆盖 (不可点击)
  CLICKABLE = 2,               // 可点击
  SELECTED = 3,                // 已选中 (在卡槽中)

## 核心算法实现

### 1. 关卡生成算法

```javascript
/**
 * 生成可通关的关卡数据
 */
function generateLevel(totalCards, typeCount, allowDeadEnd = false) {
  const cards = [];
  const groups = totalCards / 3;  // 牌组数 (每组 3 张)
  
  // 1. 生成牌组 (保证每种 3 张)
  const cardGroups = [];
  for (let i = 0; i < groups; i++) {
    const type = Math.floor(Math.random() * typeCount) + 1;
    cardGroups.push(type);
  }
  
  // 2. 展开为单牌 (每组 3 张相同)
  const cardTypes = [];
  for (const group of cardGroups) {
    cardTypes.push(group, group, group);
  }
  
  // 3. 如果是死局模式，故意制造不平衡
  if (allowDeadEnd && Math.random() < 0.7) {
    // 移除某一张牌，让该牌型只剩 2 张
    const removeIndex = Math.floor(Math.random() * cardTypes.length);
    cardTypes.splice(removeIndex, 1);
  }
  
  // 4. 打乱顺序
  shuffleArray(cardTypes);
  
  // 5. 分配到层级位置
  const positions = generateLayerPositions(totalCards);
  for (let i = 0; i < totalCards; i++) {
    cards.push({
      id: `card_${i}`,
      type: cardTypes[i],
      layer: positions[i].layer,
      row: positions[i].row,
      col: positions[i].col,
      status: CardStatus.CLICKABLE,
      rotation: (Math.random() - 0.5) * 10,
      scale: 0.95 + Math.random() * 0.1
    });
  }
  
  // 6. 计算覆盖关系
  updateCardCoverage(cards);
  
  return cards;
}
```

### 2. 覆盖关系计算

```javascript
/**
 * 更新卡牌覆盖状态
 * 规则：上层牌的顶点落在下层牌的中心区域即为覆盖
 */
function updateCardCoverage(cards) {
  for (const card of cards) {
    card.isCovered = false;
    card.status = CardStatus.CLICKABLE;
    
    // 检查是否有上层牌覆盖
    for (const other of cards) {
      if (other.layer > card.layer && isCovering(other, card)) {
        card.isCovered = true;
        card.status = CardStatus.COVERED;
        break;
      }
    }
  }
}

/**
 * 判断 cardB 是否覆盖 cardA
 * 覆盖条件：行差在 [-1, 0]，列差在 [-1, 0]
 */
function isCovering(upper, lower) {
  const rowDiff = upper.row - lower.row;
  const colDiff = upper.col - lower.col;
  
  return (rowDiff >= -1 && rowDiff <= 0) &&
         (colDiff >= -1 && colDiff <= 0);
}
```

### 3. 消除检测算法

```javascript
/**
 * 检查卡槽中的消除
 */
function checkElimination() {
  const slot = gameState.slotCards;
  
  // 1. 统计每种牌型的数量
  const typeCount = {};
  for (const card of slot) {
    typeCount[card.type] = (typeCount[card.type] || 0) + 1;
  }
  
  // 2. 查找可消除的牌型 (3 张)
  for (const [type, count] of Object.entries(typeCount)) {
    if (count === 3) {
      eliminateCards(parseInt(type));
      return true;
    }
  }
  
  return false;
}

/**
 * 消除指定牌型的卡牌
 */
function eliminateCards(type) {
  const slot = gameState.slotCards;
  const toEliminate = slot.filter(c => c.type === type);
  
  // 播放消除动画
  for (const card of toEliminate) {
    playEliminateAnimation(card);
  }
  
  // 从卡槽移除
  gameState.slotCards = slot.filter(c => c.type !== type);
  
  // 更新卡牌状态
  for (const card of toEliminate) {
    card.status = CardStatus.ELIMINATED;
  }
  
  // 重新排列卡槽
  reorderSlotCards();
  
  // 检查胜利
  checkVictory();
}
```

### 4. 失败/胜利检测

```javascript
/**
 * 检查游戏失败
 */
function checkFailure() {
  const slot = gameState.slotCards;
  
  if (slot.length < GAME_CONFIG.SLOT_CAPACITY) {
    return false;
  }
  
  if (checkElimination()) {
    return false;
  }
  
  gameState.isGameOver = true;
  showGameOverPanel();
  return true;
}

/**
 * 检查游戏胜利
 */
function checkVictory() {
  const remainingCards = gameState.cards.filter(
    c => c.status !== CardStatus.ELIMINATED
  );
  
  if (remainingCards.length === 0) {
    gameState.isLevelComplete = true;
    showVictoryPanel();
    reportLevelComplete(gameState.currentLevel);
    return true;
  }
  
  return false;
}
```

## 道具系统实现

### 道具配置

| 道具 | 功能 | 获取方式 | 使用逻辑 |
|------|------|----------|----------|
| **移出** | 移除卡槽前 3 张牌 | 看广告/分享 | 返回牌堆，不改变状态 |
| **洗牌** | 打乱所有未消除的牌 | 看广告/分享 | 重新随机分配类型 |
| **撤回** | 撤销上一步操作 | 看广告/分享 | 从卡槽返回原位置 |
| **复活** | 失败后清空卡槽 | 看广告/分享 | 清空卡槽，继续游戏 |

### 道具使用代码

```javascript
// 移出道具
function useRemoveProp() {
  if (gameState.props.remove <= 0) return;
  const toRemove = gameState.slotCards.slice(0, 3);
  
  for (const card of toRemove) {
    animateReturnToDeck(card);
    card.status = CardStatus.COVERED;
  }
  
  gameState.slotCards = gameState.slotCards.slice(3);
  gameState.props.remove--;
  checkElimination();
}

// 洗牌道具
function useShuffleProp() {
  const remaining = gameState.cards.filter(
    c => c.status !== CardStatus.ELIMINATED && c.status !== CardStatus.SELECTED
  );
  
  const types = remaining.map(c => c.type);
  shuffleArray(types);
  
  remaining.forEach((card, i) => {
    card.type = types[i];
    updateCardVisual(card);
  });
  
  gameState.props.shuffle--;
}
```
