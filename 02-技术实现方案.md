### 1.2 技术选型

| 模块 | 技术 | 说明 |
|------|------|------|
| **前端框架** | 原生微信小程序 | 无需额外框架 |
| **游戏引擎** | 无（纯 DOM 实现） | 简单游戏不需要游戏引擎 |
| **语言** | JavaScript (ES6+) | 微信小程序支持 |
| **样式** | WXSS | 类似 CSS |
| **数据存储** | wx.setStorage | 本地存储 |
| **后端** | 无（或微信云开发） | 初期不需要后端 |
| **广告** | 微信广告 SDK | 官方集成 |

### 1.3 为什么不使用游戏引擎？

```
✅ 优势：
- 零学习成本（会 JS 就能写）
- 包体积小（<1MB）
- 加载速度快
- 适合简单消除游戏

❌ 不适用游戏引擎的原因：
- Cocos/Unity 包体积大（>5MB）
- 学习成本高
- 对于堆叠消除这种简单游戏，性能过剩
```

## 三、核心模块实现

### 3.1 游戏主逻辑 (game.js)

```javascript
// pages/game/game.js
const GameState = require('../../utils/GameState');
const { generateLevel } = require('../../utils/generator');
const AdManager = require('../../utils/ad');

Page({
  data: {
    currentLevel: 1,
    cards: [],
    slotCards: [],
    props: {
      remove: 1,
      shuffle: 1,
      undo: 1,
      revive: 1
    },
    isGameOver: false,
    isLevelComplete: false
  },

  onLoad() {
    // 初始化游戏
    this.game = new GameState();
    this.adManager = new AdManager();
    
    // 加载关卡
    this.loadLevel(1);
    
    // 开始计时
    this.startTimer();
  },

  // 加载关卡
  loadLevel(level) {
    const cards = generateLevel(level);
    
    this.setData({
      currentLevel: level,
      cards: cards.map(c => ({
        ...c,
        imagePath: this.getCardImagePath(c.type)
      }))
    });
  },

  // 卡牌点击
  onCardTap(e) {
    const cardId = e.currentTarget.dataset.id;
    const card = this.game.cards.find(c => c.id === cardId);
    
    if (!card || !card.isClickable()) {
      // 播放抖动动画
      this.shakeCard(cardId);
      return;
    }
    
    // 添加到卡槽
    const success = this.game.addToSlot(card);
    
    if (success) {
      this.updateUI();
      
      // 播放音效
      this.playAudio('click');
    }
  },

  // 使用道具
  onUseProp(e) {
    const propType = e.currentTarget.dataset.type;
    
    if (this.data.props[propType] <= 0) {
      // 道具不足，提示看广告
      this.adManager.watchAdForProp(propType);
      return;
    }
    
    // 使用道具
    this.useProp(propType);
  },

  // 复活
  onRevive() {
    if (this.data.props.revive > 0) {
      this.useProp('revive');
    } else {
      // 看广告复活
      this.adManager.watchAdForRevive();
    }
  },

  // 重新开始
  onRestart() {
    this.loadLevel(this.data.currentLevel);
    this.setData({ isGameOver: false });
  },

  // 下一关
  onNextLevel() {
    const nextLevel = this.data.currentLevel + 1;
    if (nextLevel <= 2) {
      this.loadLevel(nextLevel);
      this.setData({ isLevelComplete: false });
    } else {
      wx.showToast({
        title: '已是最后一关',
        icon: 'none'
      });
    }
  },

  // 更新 UI
  updateUI() {
    this.setData({
      cards: this.game.cards.map(c => ({
        ...c,
        imagePath: this.getCardImagePath(c.type)
      })),
      slotCards: this.game.slotCards.map(c => ({
        ...c,
        imagePath: this.getCardImagePath(c.type)
      })),
      props: this.game.props,
      isGameOver: this.game.isGameOver,
      isLevelComplete: this.game.isLevelComplete
    });
  },

  // 获取卡牌图片路径
  getCardImagePath(type) {
    const paths = {
      1: '/assets/images/cards/sunflower.png',
      2: '/assets/images/cards/clover.png',
      3: '/assets/images/cards/flowerpot.png',
      4: '/assets/images/cards/gloves.png',
      5: '/assets/images/cards/waterdrop.png',
      6: '/assets/images/cards/stump.png',
      7: '/assets/images/cards/vine.png',
      8: '/assets/images/cards/stone.png',
      9: '/assets/images/cards/sprout.png',
      10: '/assets/images/cards/butterfly.png',
      11: '/assets/images/cards/bee.png',
      12: '/assets/images/cards/mushroom.png'
    };
    return paths[type] || paths[1];
  },

  // 播放音效
  playAudio(type) {
    const audioContext = wx.createInnerAudioContext();
    audioContext.src = `/assets/audio/sfx/${type}.mp3`;
    audioContext.play();
  },

  // 开始计时
  startTimer() {
    this.timer = setInterval(() => {
      this.game.elapsedTime++;
    }, 1000);
  },

  // 抖动动画
  shakeCard(cardId) {
    // 使用微信小程序动画 API
    const animation = wx.createAnimation({
      duration: 300,
      timingFunction: 'ease-in-out'
    });
    
    animation.translateX(-5).step()
      .translateX(5).step()
      .translateX(0).step();
    
    this.setData({
      [`cards[${cardId}].animationData`]: animation.export()
    });
  }
});
```

### 3.2 关卡生成算法 (generator.js)

```javascript
// utils/generator.js

const LEVEL_CONFIG = {
  1: {
    totalCards: 24,
    cardTypes: 4,
    layers: 3,
    deadEndChance: 0
  },
  2: {
    totalCards: 144,
    cardTypes: 12,
    layers: 10,
    deadEndChance: 0.7
  }
};

/**
 * 生成关卡
 */
function generateLevel(level) {
  const config = LEVEL_CONFIG[level];
  const cards = [];
  
  // 1. 生成牌组（每组 3 张）
  const groups = config.totalCards / 3;
  const cardTypes = [];
  
  for (let i = 0; i < groups; i++) {
    const type = Math.floor(Math.random() * config.cardTypes) + 1;
    cardTypes.push(type, type, type);
  }
  
  // 2. 死局模式：70% 概率移除一张
  if (config.deadEndChance > 0 && Math.random() < config.deadEndChance) {
    cardTypes.splice(Math.floor(Math.random() * cardTypes.length), 1);
  }
  
  // 3. 打乱
  shuffleArray(cardTypes);
  
  // 4. 分配到位置
  const positions = generatePositions(config);
  for (let i = 0; i < cardTypes.length; i++) {
    const pos = positions[i];
    const screenPos = toScreen(pos);
    
    cards.push({
      id: `card_${i}`,
      type: cardTypes[i],
      layer: pos.layer,
      row: pos.row,
      col: pos.col,
      x: screenPos.x,
      y: screenPos.y,
      zIndex: 100 + pos.layer * 10,
      status: 2, // CLICKABLE
      isCovered: false,
      rotation: (Math.random() - 0.5) * 10,
      scale: 0.95 + Math.random() * 0.1
    });
  }
  
  // 5. 计算覆盖
  updateCoverage(cards);
  
  return cards;
}

/**
 * 生成位置
 */
function generatePositions(config) {
  const positions = [];
  const grids = getLayerGrids(config.layers);
  
  grids.forEach((grid, layerIndex) => {
    grid.forEach((row, rowIndex) => {
      row.forEach((hasCard, colIndex) => {
        if (hasCard === 1) {
          positions.push({
            layer: layerIndex + 1,
            row: rowIndex,
            col: colIndex
          });
        }
      });
    });
  });
  
  return positions;
}

/**
 * 层级转屏幕坐标
 */
function toScreen(pos) {
  const baseX = 150;
  const baseY = 200;
  const CARD_WIDTH = 48;
  const CARD_HEIGHT = 60;
  const OFFSET_X = 24;
  const OFFSET_Y = -12;
  
  return {
    x: baseX + pos.col * CARD_WIDTH - pos.layer * OFFSET_X,
    y: baseY + pos.row * CARD_HEIGHT - pos.layer * OFFSET_Y
  };
}

/**
 * 更新覆盖状态
 */
function updateCoverage(cards) {
  cards.forEach(card => {
    card.isCovered = false;
    card.status = 2;
    
    cards.forEach(other => {
      if (other.layer > card.layer && isCovering(other, card)) {
        card.isCovered = true;
        card.status = 1; // COVERED
      }
    });
  });
}

/**
 * 判断覆盖
 */
function isCovering(upper, lower) {
  const rowDiff = upper.row - lower.row;
  const colDiff = upper.col - lower.col;
  
  return (rowDiff >= -1 && rowDiff <= 0) &&
         (colDiff >= -1 && colDiff <= 0);
}

/**
 * 洗牌算法
 */
function shuffleArray(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }
}

/**
 * 获取层级网格（简化版）
 */
function getLayerGrids(layers) {
  // 实际应该根据关卡配置返回
  // 这里简化处理
  const grids = [];
  for (let i = 0; i < layers; i++) {
    const size = Math.max(2, 5 - Math.floor(i / 2));
    const grid = [];
    for (let r = 0; r < size; r++) {
      const row = [];
      for (let c = 0; c < size; c++) {
        row.push(1);
      }
      grid.push(row);
    }
    grids.push(grid);
  }
  return grids;
}

module.exports = { generateLevel, LEVEL_CONFIG };
```

### 3.3 广告管理 (ad.js)

```javascript
// utils/ad.js

class AdManager {
  constructor() {
    this.adUnits = {
      rewardVideo: 'adunit-xxxxxxxxxxxxx',
      interstitial: 'adunit-yyyyyyyyyyyyy'
    };
    this.rewardedVideoAd = null;
    this.pendingReward = null;
    
    this.init();
  }
  
  init() {
    if (wx.createRewardedVideoAd) {
      this.rewardedVideoAd = wx.createRewardedVideoAd({
        adUnitId: this.adUnits.rewardVideo
      });
      
      this.rewardedVideoAd.onClose((res) => {
        if (res && res.isEnded) {
          this.grantReward();
        }
      });
    }
  }
  
  watchAdForProp(propType) {
    this.pendingReward = { type: 'prop', propType, count: 1 };
    this.rewardedVideoAd?.show();
  }
  
  watchAdForRevive() {
    this.pendingReward = { type: 'revive' };
    this.rewardedVideoAd?.show();
  }
  
  grantReward() {
    const pages = getCurrentPages();
    const page = pages[pages.length - 1];
    
    if (this.pendingReward.type === 'prop') {
      page.game.addProp(this.pendingReward.propType, 1);
      wx.showToast({ title: '获得道具！', icon: 'success' });
    } else if (this.pendingReward.type === 'revive') {
      page.game.revive();
      wx.showToast({ title: '复活成功！', icon: 'success' });
    }
    
    this.pendingReward = null;
  }
}

module.exports = AdManager;
```
